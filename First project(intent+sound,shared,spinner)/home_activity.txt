package com.example.myapplication

import android.app.Activity
import android.content.Intent
import android.os.Bundle
import android.view.View
import android.widget.Button
import android.widget.ImageView
import android.widget.TextView
import android.widget.Toast
import androidx.activity.enableEdgeToEdge
import androidx.appcompat.app.AppCompatActivity
import androidx.core.net.toUri
import androidx.core.view.ViewCompat
import androidx.core.view.WindowInsetsCompat
import com.github.barteksc.pdfviewer.PDFView

class HomeActivity : AppCompatActivity() {
    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        enableEdgeToEdge()
        setContentView(R.layout.activity_home)
        ViewCompat.setOnApplyWindowInsetsListener(findViewById(R.id.main)) { v, insets ->
            val systemBars = insets.getInsets(WindowInsetsCompat.Type.systemBars())
            v.setPadding(systemBars.left, systemBars.top, systemBars.right, systemBars.bottom)
            insets
        }
        Toast.makeText(this, "Welcome to Page2 :D", Toast.LENGTH_SHORT).show()
        val text1:TextView=findViewById(R.id.text)
        val text2:TextView=findViewById(R.id.text1)
        val et=intent.getStringExtra("text")
        val user=intent.getStringExtra("username")
        text1.text=et
        text2.text="Hello ${user} :D"
        val btn1:Button=findViewById(R.id.btn)
        val btn2:Button=findViewById(R.id.btn1)
        btn1.setOnClickListener {
            val website="https://www.facebook.com".toUri()
            val i=Intent(Intent.ACTION_VIEW,website)
            startActivity(i)
            finish()
        }
        btn2.setOnClickListener {
            val i=Intent(this,PageActivity::class.java)
            startActivity(i)
            finish()
        }
        val text3:TextView=findViewById(R.id.text2)
        val text4:TextView=findViewById(R.id.text3)
        text3.setOnClickListener {
            val i=Intent(Intent.ACTION_GET_CONTENT)
            i.type="image/*"
            startActivityForResult(i,22)
        }
        text4.setOnClickListener {
            val i=Intent(Intent.ACTION_OPEN_DOCUMENT)
            i.addCategory(Intent.CATEGORY_OPENABLE)
            i.type="application/pdf"
            startActivityForResult(i,20)
        }
    }

    override fun onActivityResult(requestCode: Int, resultCode: Int, data: Intent?) {
        super.onActivityResult(requestCode, resultCode, data)
        if (resultCode== Activity.RESULT_OK && data!=null){
            val uri = data.data
            val img: ImageView = findViewById(R.id.image)
            val pdfView: PDFView = findViewById(R.id.pdf)
            if (requestCode == 22 && uri != null) {
                img.visibility = View.VISIBLE
                pdfView.visibility = View.GONE
                img.setImageURI(uri)
            }
            else if(requestCode == 20 && uri != null) {
                contentResolver.takePersistableUriPermission(
                    uri,
                    Intent.FLAG_GRANT_READ_URI_PERMISSION
                )
                img.visibility = View.GONE
                pdfView.visibility = View.VISIBLE
                pdfView.fromUri(uri)
                    .enableSwipe(true)
                    .swipeHorizontal(false)
                    .enableDoubletap(true)
                    .load()
            }
        }
    }
}
